class DotReduction(Layer):
    def call(self, inputs):
        u, i = inputs
        return tf.reduce_sum(u * i, axis=1)
        
class RecommendationModel:
    def __init__(self, embedding_dim=64):
        self.embedding_dim = embedding_dim
        self.user_to_idx = user_to_idx
        self.item_to_idx = item_to_idx
        self.valid_items = valid_items
        self.products_clean = products_clean
        self.skin_encoder = skin_encoder
        self.num_skin_types = num_skin_types
        
        self.user_emb = tf.keras.layers.Embedding(len(user_to_idx) + 1, embedding_dim)
        self.item_emb = tf.keras.layers.Embedding(len(item_to_idx) + 1, embedding_dim)
        self.skin_emb = tf.keras.layers.Embedding(self.num_skin_types + 1, embedding_dim // 2)  
        
        inputs = tf.keras.Input(shape=(3,))  
        user_out = self.user_emb(inputs[:,0])
        item_out = self.item_emb(inputs[:,1])
        skin_out = self.skin_emb(inputs[:,2])
        combined_user = tf.keras.layers.Concatenate()([user_out, skin_out])  
        
        combined_user = Dense(self.embedding_dim)(combined_user)
        
        outputs = DotReduction()([combined_user, item_out])
        self.cf_model = tf.keras.Model(inputs=inputs, outputs=outputs)
        self.cf_model.compile(optimizer='adam', loss='mse')
        
        self.tfidf = TfidfVectorizer(stop_words='english')
        self.tfidf_matrix = self.tfidf.fit_transform(self.products_clean['text_profile'])

    def train(self, data, epochs=5, batch_size=4096):
        train_inputs = np.stack([data['user_idx'], data['item_idx'], data['skin_idx']], axis=1)
        train_labels = data['rating'].values.astype(np.float32)
        
        self.cf_model.fit(train_inputs, train_labels, batch_size=batch_size, epochs=epochs, verbose=1)
        print(" ")
        
    def get_cf_scores(self, user_id):
        if user_id not in self.user_to_idx:
            return np.array([])
        
        user_idx = self.user_to_idx[user_id]
        skin_type = data[data['author_id'] == user_id]['skin_type'].iloc[0]
        skin_idx = self.skin_encoder.transform([skin_type])[0]
        
        item_idxs = np.arange(len(self.valid_items))
        user_idxs = np.full(len(self.valid_items), user_idx)
        skin_idxs = np.full(len(self.valid_items), skin_idx)
        inputs = np.stack([user_idxs, item_idxs, skin_idxs], axis=1)
        scores = self.cf_model.predict(inputs, batch_size=4096).flatten()
        return scores
    
    def get_cbf_scores(self, user_id):
        user_liked = data[(data['author_id'] == user_id) & (data['positive'] == 1)]['item_idx'].values
        if len(user_liked) == 0:
            return np.zeros(len(self.valid_items))
        
        liked_vectors = self.tfidf_matrix[user_liked]
        user_profile = np.asarray(liked_vectors.mean(axis=0))
        
        skin_type = data[data['author_id'] == user_id]['skin_type'].iloc[0]
        skin_boost = np.array([1.2 if skin_type.lower() in str(row['highlights'] + row['secondary_category']).lower() else 1.0 
                               for _, row in self.products_clean.iterrows()])
        
        sim_scores = cosine_similarity(user_profile, self.tfidf_matrix).flatten() * skin_boost
        return sim_scores
    
    def get_hybrid_recommendations(self, user_id, top_n=10, weight_cf=0.2, weight_cbf=0.9):
        if user_id not in self.user_to_idx:
            print("  ")
            return pd.DataFrame()

        cf_scores = self.get_cf_scores(user_id)          
        cbf_scores = self.get_cbf_scores(user_id)       

        cf_scores = (cf_scores - cf_scores.min()) / (cf_scores.max() - cf_scores.min() + 1e-8)
        cbf_scores = (cbf_scores - cbf_scores.min()) / (cbf_scores.max() - cbf_scores.min() + 1e-8)

        hybrid_scores = weight_cf * cf_scores + weight_cbf * cbf_scores

        top_indices = np.argsort(hybrid_scores)[-top_n:][::-1]  # descending

        recs = []
        for idx in top_indices:
            pid = self.valid_items[idx]
            row = self.products_clean[self.products_clean['product_id'] == pid].iloc[0]
            recs.append({
                "product_id": pid,
                "": row['product_name'],
                "": row['brand_name'],
                "": row['price_usd'],
                "": round(row['rating'], 2),
                "hybrid_score": float(hybrid_scores[idx])
            })

        return pd.DataFrame(recs)
    
    def get_safe_hybrid_recommendations(self, user_id, blacklist=["alcohol", "fragrance", "sulfate"], top_n=10):
        hybrid_df = self.get_hybrid_recommendations(user_id, top_n=top_n*5)
        
        safe = []
        for _, row in hybrid_df.iterrows():
            pid = row['product_id']
            ingredients = self.products_clean[self.products_clean['product_id'] == pid]['ingredients'].iloc[0]
            if any(b.lower() in str(ingredients).lower() for b in blacklist):
                continue
            safe.append(row.to_dict())
            if len(safe) >= top_n: break
        
        return pd.DataFrame(safe)

def evaluate_metrics(model, test_data, top_n=10, min_relevant=3):
    users = test_data['author_id'].unique()
    selected_users = np.random.choice(users, size=1000, replace=False) 
    processed_users = 0
    all_recs = []
    precisions = []
    recalls = []
    ndcgs = []
    
    for user_id in selected_users:
        true_relevant = test_data[(test_data['author_id'] == user_id) & (test_data['positive'] == 1)]['product_id'].values
        if len(true_relevant) < min_relevant: continue
        
        recs_df = model.get_hybrid_recommendations(user_id, top_n=top_n)
        pred_items = recs_df['product_id'].values
        
        relevant_in_top = np.isin(pred_items, true_relevant).sum()
        precision = relevant_in_top / top_n
        precisions.append(precision)
        
        recall = relevant_in_top / len(true_relevant)
        recalls.append(recall)
        
        true_scores = np.isin(pred_items, true_relevant).astype(int)
        ideal_scores = np.sort(true_scores)[::-1]
        if np.sum(true_scores) > 0:
            ndcg = ndcg_score([ideal_scores], [true_scores])
            ndcgs.append(ndcg)
        
        all_recs.extend(pred_items)
        processed_users += 1
    
    print(f"Processed {processed_users} users with >= {min_relevant} relevant items")
    
    unique_recs = set(all_recs)
    coverage = len(unique_recs) / len(valid_items) * 100 if valid_items else 0
    
    if processed_users > 1:
        rec_vectors = []
        for user_id in users[:processed_users]:  
            recs_df = model.get_hybrid_recommendations(user_id, top_n=top_n)
            rec_vec = np.zeros(len(valid_items))
            for pid in recs_df['product_id']:
                item_idx = item_to_idx.get(pid, 0)
                rec_vec[item_idx] = 1
            rec_vectors.append(rec_vec)
        
        sim_matrix = cosine_similarity(rec_vectors)
        avg_sim = np.mean(sim_matrix[np.triu_indices(processed_users, k=1)])
        personalization = (1 - avg_sim) * 100
    else:
        personalization = 100.0
    
    metrics = {
        'Precision': np.mean(precisions) if precisions else 0.0,
        'Recall': np.mean(recalls) if recalls else 0.0,
        'NDCG': np.mean(ndcgs) if ndcgs else 0.0,
        'Coverage': coverage,
        'Personalization': personalization
    }
    
    return metrics
